        Solution from markscheme 
        

 
        FUNCTION Factorial(ThisNum : INTEGER) RETURNS INTEGER
          DECLARE Value : INTEGER
          IF ThisNum < 0 THEN
            Value <- -1
          ELSE
            Value <-  1
            WHILE ThisNum <> 0 DO
              Value <- Value * ThisNum
              ThisNum <- ThisNum -1
            ENDWHILE
          ENDIF
          RETURN Value
        ENDFUNCTION 
       


    

    

        Alternative Solution 
        

 
          FUNCTION Factorial(ThisNum : INTEGER) RETURNS INTEGER
            IF ThisNum > 1 THEN
              RETURN ThisNum * Factorial(ThisNum - 1)
            ELSE
              IF ThisNum = 1 OR ThisNum = 0 THEN
                RETURN 1
              ELSE
                RETURN -1
              ENDIF
            ENDIF
          ENDFUNCTION

        Idea behind solution in plain English 
        
1)Initialise a variable as an empty string 

        2)If count is less than 1 assign error to the variable

        3)In a loop that goes count amount of times, add the character to the previously initialized variable

        4)Return the variable

    


    

        Solution from markscheme 
        
FUNCTION MakeString(Count : INTEGER, AChar : CHAR) RETURNS STRING 
          	DECLARE MyString : STRING

          	DECLARE Index : INTEGER

          	IF Count < 1 THEN

          		MyString <- "ERROR"

          	ELSE

          		MyString <- ""

          		FOR Index <- 1 TO Count

          			MyString <- MyString & AChar

          		NEXT Index

          	ENDIF

          	RETURN MyString

          ENDFUNCTION

          Solution from markscheme 
          

 
          Function Status(Actual: INTEGER, Min: INTEGER, Max : INTEGER) RETURNS CHAR
            DECLARE Result : CHAR
            CONSTANT Accept = 'A'
            CONSTANT Reject = 'R'
            CONSTANT ReCheck = 'C'
            Result <- ReCheck
            //Check if reject
            IF Actual > Max + 2 OR Actual < Min - 2 THEN
              Result <- Reject
            ENDIF
            //Check if acceptable
            IF Actual < Max - 2 AND Actual > Min + 2 THEN
              Result <- Accept
            ENDIF
            RETURN Result
          ENDFUNCTION
